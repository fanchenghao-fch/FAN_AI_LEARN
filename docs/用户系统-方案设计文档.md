### 关键设计决策

| 决策 | 选型 | 原因 |
|------|------|------|
| 数据库 | MySQL 8.0 | 用户指定，配置简单 |
| 异步驱动 | asyncmy | SQLAlchemy 2.0 官方推荐 MySQL 异步驱动 |
| ORM | SQLAlchemy 2.0 async | 与 FastAPI async/await 生态完美融合 |
| 鉴权 | JWT (HS256) | 轻量、无状态、7天过期 |
| 微信登录 | wx.login → code → openid | 微信官方流程 |
| 前端状态 | Zustand | 与现有 store 保持一致 |
| Mock 登录 | 昵称匹配 | 仅开发/演示用，不设密码 |

#### 2.2.1 `users` — 用户表

```sql
CREATE TABLE users (
    id          CHAR(36)      NOT NULL PRIMARY KEY,        -- UUID
    open_id     VARCHAR(64)   DEFAULT NULL,                 -- 微信 openid
    union_id    VARCHAR(64)   DEFAULT NULL,                 -- 微信 unionid
    nickname    VARCHAR(64)   NOT NULL DEFAULT '灯灯学员',   -- 昵称
    avatar_url  VARCHAR(512)  DEFAULT NULL,                 -- 头像 URL
    coins       INT           NOT NULL DEFAULT 0,           -- 金币
    experience  INT           NOT NULL DEFAULT 0,           -- 经验值
    level_id    INT           NOT NULL DEFAULT 1,           -- 等级 (FK → level_configs.id)
    created_at  DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME      NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uk_open_id (open_id),
    INDEX idx_nickname (nickname),
    INDEX idx_experience (experience)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

```python
"""Async SQLAlchemy engine and session factory for MySQL via asyncmy."""

from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from app.config import settings

engine = create_async_engine(
    settings.DATABASE_URL,
    echo=False,          # 生产环境关闭 SQL 日志
    pool_size=10,        # 连接池大小
    max_overflow=20,     # 超出 pool_size 的最大连接数
    pool_recycle=3600,   # 1小时回收连接
    pool_pre_ping=True,  # 连接前检测有效性
)

AsyncSessionLocal = async_sessionmaker(
    engine,
    class_=AsyncSession,
    expire_on_commit=False,
)


async def get_db() -> AsyncSession:
    """FastAPI dependency: provide async DB session per request."""
    async with AsyncSessionLocal() as session:
        try:
            yield session
            await session.commit()
        except Exception:
            await session.rollback()
            raise
```

### 3.4 认证系统

#### `app/auth.py` — JWT + 依赖注入

```python
"""JWT token creation, verification, and FastAPI dependency."""

from datetime import datetime, timedelta
from typing import Optional
import jwt  # PyJWT

from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

from app.config import settings
from app.database import get_db
from app.models.orm import User

bearer_scheme = HTTPBearer(auto_error=False)


def create_access_token(user_id: str) -> str:
    """Create a JWT access token with 7-day expiry."""
    payload = {
        "sub": user_id,
        "iat": datetime.utcnow(),
        "exp": datetime.utcnow() + timedelta(days=settings.JWT_EXPIRE_DAYS),
    }
    return jwt.encode(payload, settings.JWT_SECRET, algorithm=settings.JWT_ALGORITHM)


def decode_access_token(token: str) -> Optional[str]:
    """Decode JWT and return user_id, or None if invalid/expired."""
    try:
        payload = jwt.decode(
            token, settings.JWT_SECRET, algorithms=[settings.JWT_ALGORITHM]
        )
        return payload.get("sub")
    except (jwt.ExpiredSignatureError, jwt.InvalidTokenError):
        return None


async def get_current_user(
    credentials: Optional[HTTPAuthorizationCredentials] = Depends(bearer_scheme),
    db: AsyncSession = Depends(get_db),
) -> Optional[User]:
    """
    FastAPI dependency: extract current user from JWT Bearer token.
    
    Returns None if no token provided (unauthenticated),
    raises 401 if token is invalid/expired.
    """
    if credentials is None:
        return None  # 允许未登录访问 quiz 接口

    user_id = decode_access_token(credentials.credentials)
    if user_id is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Token 已过期或无效，请重新登录",
        )

    result = await db.execute(select(User).where(User.id == user_id))
    user = result.scalar_one_or_none()

    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="用户不存在",
        )

    return user


async def require_user(
    user: Optional[User] = Depends(get_current_user),
) -> User:
    """
    FastAPI dependency: requires authenticated user. Raises 401 if not logged in.
    Use for /api/user/* endpoints.
    """
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="请先登录",
        )
    return user
```

### 3.6 积分计算服务

#### `app/services/points.py`

```python
"""Points/coins calculation according to product rules."""

from dataclasses import dataclass


@dataclass
class PointsInput:
    score: int            # 答对数
    total: int            # 总题数
    combo_max: int        # 最高连击
    is_first_today: bool  # 是否今日首次闯关


MAX_COINS_PER_QUIZ = 100


def calculate_coins(input: PointsInput) -> int:
    """
    积分规则：
      - 基础：+10（完成一次闯关）
      - 正确：+2 × 答对数
      - Combo ≥5：+5
      - Combo ≥10：+15
      - 每日首胜：+20
    上限：100 金币/次
    """
    coins = 10                     # 基础
    coins += input.score * 2       # 答对奖励

    if input.combo_max >= 10:
        coins += 15
    elif input.combo_max >= 5:
        coins += 5

    if input.is_first_today:
        coins += 20

    return min(coins, MAX_COINS_PER_QUIZ)


def calculate_experience(score: int, total: int) -> int:
    """经验值 = 答对数 × 10 + 闯关完成 × 5"""
    return score * 10 + 5
```

#### `stores/userStore.ts`（新建）

```typescript
/**
 * Zustand store for user authentication and profile state.
 */

import { create } from "zustand";
import type { UserProfile, UserStats, LevelInfo } from "../types/user";

// ── Token persistence helpers ──────────────────────────

const TOKEN_KEY = "quiz_user_token";
const PROFILE_CACHE_KEY = "quiz_user_profile";

function getStoredToken(): string | null {
  try {
    return localStorage.getItem(TOKEN_KEY);
  } catch {
    return null;
  }
}

function setStoredToken(token: string | null): void {
  try {
    if (token) {
      localStorage.setItem(TOKEN_KEY, token);
    } else {
      localStorage.removeItem(TOKEN_KEY);
      localStorage.removeItem(PROFILE_CACHE_KEY);
    }
  } catch {
    // 小程序环境 localStorage 不可用 → Taro.setStorageSync
  }
}

// ── Store ──────────────────────────────────────────────

interface UserState {
  // Auth
  token: string | null;
  isLoggedIn: boolean;
  isLoginPending: boolean;

  // Profile
  profile: UserProfile | null;
  stats: UserStats | null;

  // Login modal
  showLoginModal: boolean;
}

interface UserActions {
  // Auth actions
  mockLogin: (nickname: string, avatarUrl?: string) => Promise<void>;
  wechatLogin: () => Promise<void>;
  logout: () => void;

  // Profile actions
  fetchProfile: () => Promise<void>;
  fetchStats: () => Promise<void>;  // lightweight

  // UI
  setShowLoginModal: (show: boolean) => void;

  // Computed
  getLevelProgress: () => number; // 0-1
}

export const useUserStore = create<UserState & UserActions>((set, get) => ({
  // Initial state
  token: getStoredToken(),
  isLoggedIn: !!getStoredToken(),
  isLoginPending: false,
  profile: null,
  stats: null,
  showLoginModal: false,

  // ── Auth ──────────────────────────────────────────

  mockLogin: async (nickname, avatarUrl) => {
    set({ isLoginPending: true });
    try {
      const res = await authApi.mockLogin({ nickname, avatar_url: avatarUrl });
      setStoredToken(res.token);
      set({
        token: res.token,
        isLoggedIn: true,
        isLoginPending: false,
        showLoginModal: false,
      });
      // Prefetch profile after login
      get().fetchProfile();
    } catch (e) {
      set({ isLoginPending: false });
      throw e;
    }
  },

  wechatLogin: async () => {
    set({ isLoginPending: true });
    try {
      // 1. wx.login → code
      const loginRes = await new Promise<{ code: string }>((resolve, reject) => {
        wx.login({
          success: (res) => resolve(res),
          fail: reject,
        });
      });
      // 2. 后端换取 token
      const res = await authApi.wechatLogin({ code: loginRes.code });
      setStoredToken(res.token);
      set({
        token: res.token,
        isLoggedIn: true,
        isLoginPending: false,
        showLoginModal: false,
      });
      get().fetchProfile();
    } catch (e) {
      set({ isLoginPending: false });
      throw e;
    }
  },

  logout: () => {
    setStoredToken(null);
    set({
      token: null,
      isLoggedIn: false,
      profile: null,
      stats: null,
    });
  },

  // ── Profile ───────────────────────────────────────

  fetchProfile: async () => {
    const { token } = get();
    if (!token) return;
    try {
      const res = await userApi.getProfile(token);
      set({ profile: res.user, stats: res.stats });
    } catch (e) {
      // 401 → 自动登出
      if ((e as { status?: number }).status === 401) {
        get().logout();
      }
    }
  },

  fetchStats: async () => {
    const { token } = get();
    if (!token) return;
    try {
      const res = await userApi.getStats(token);
      set({ stats: res });
    } catch {
      // silent
    }
  },

  // ── UI ────────────────────────────────────────────
  setShowLoginModal: (show) => set({ showLoginModal: show }),

  // ── Computed ──────────────────────────────────────
  getLevelProgress: () => {
    const { profile } = get();
    if (!profile) return 0;
    const { level } = profile;
    if (level.next_level_exp <= level.min_exp) return 1;
    return Math.min(
      1,
      (profile.experience - level.min_exp) /
        (level.next_level_exp - level.min_exp)
    );
  },
}));
```

## 8. 风险与边界

### 8.1 风险

| 风险 | 等级 | 应对 |
|------|------|------|
| 微信 `WECHAT_APP_ID/SECRET` 未配置 | 中 | H5 Mock 登录先行；微信登录接口返回 503 提示配置 |
| asyncmy 与 Python 3.12 兼容性 | 低 | asyncmy 0.2.9+ 已支持 Python 3.12 |
| 小程序 localStorage 不可用 | 中 | 降级为 `Taro.setStorageSync` / `Taro.getStorageSync` |
| 并发答题记录保存 | 低 | 当前为单用户模式，无并发压力 |

### 8.2 边界

- **不修改现有答题逻辑**：quiz 生成/校验/分析链条不变，只在 analyze 接口末尾"附加"保存逻辑
- **不破坏未登录用户流程**：`get_current_user` 依赖返回 `Optional[User]`，quiz 接口对未登录用户透明
- **前端不引入新依赖**：Zustand 已有，Taro 导航/存储使用官方 API
- **MySQL 版本**：目标 8.0，utf8mb4 字符集
- **积分规则可调**：`app/services/points.py` 的常量（10/2/5/15/20/100）后续可改为配置化
